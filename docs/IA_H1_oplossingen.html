Afsluitende Opdracht 1.2 Nederlandse vlag

import turtle

# NAAR BEGINPOSITIE:
turtle.penup()
turtle.goto(0, 0)
turtle.pendown()

# TEKEN RODE VLAK:
turtle.color("red")
turtle.begin_fill()
turtle.forward(200)
turtle.right(90)
turtle.forward(40)
turtle.right(90)
turtle.forward(200)
turtle.right(90)
turtle.forward(40)
turtle.right(90)
turtle.end_fill()

# TEKEN WITTE VLAK
turtle.penup()
turtle.goto(0, -40)
turtle.pendown()

turtle.color("white")
turtle.begin_fill()
turtle.forward(200)
turtle.right(90)
turtle.forward(40)
turtle.right(90)
turtle.forward(200)
turtle.right(90)
turtle.forward(40)
turtle.right(90)
turtle.end_fill()

# TEKEN BLAUWE VLAK:
turtle.penup()
turtle.goto(0, -80)
turtle.pendown()

turtle.color("blue")
turtle.begin_fill()
turtle.forward(200)
turtle.right(90)
turtle.forward(40)
turtle.right(90)
turtle.forward(200)
turtle.right(90)
turtle.forward(40)
turtle.right(90)
turtle.end_fill()

# OMLIJNING VAN ROOD-WIT-BLAUW VLAKKEN:
turtle.penup()
turtle.goto(0, 0)
turtle.pendown()
turtle.color("black")
turtle.forward(200)
turtle.right(90)
turtle.forward(120)
turtle.right(90)
turtle.forward(200)
turtle.right(90)
turtle.forward(120)

# TEKEN MAST:
turtle.right(180)
turtle.forward(250)

turtle.done()
