Opgave 14.1 Meerdere waarden teruggeven

import random

jaarlaag = [4,5,6]
klas = ["a", "b", "c", "d"]

def kiesWillekeurigeKlas():
  willekeurige_jaarlaag = random.choice( jaarlaag )
  willekeurige_klas = random.choice( klas )
  return willekeurige_jaarlaag, willekeurige_klas

### HOOFDPROGRAMMA
jaar, klas = kiesWillekeurigeKlas()
print(jaar)
print(klas)

Opgave 14.2 Prijs inclusief BTW

### GLOBALE VARIABELEN
BTW = 0.21 # 21% BTW

### FUNCTIE DEFINITIES
def berekenPrijsInclBTW(prijs_zonder_BTW):
	prijs_incl_BTW = prijs_zonder_BTW * (1+BTW) #vermenigvuligen met 1,21
	afgeronde_prijs_incl_BTW = round(prijs_incl_BTW,2)
	return afgeronde_prijs_incl_BTW

### HOOFDPROGRAMMA
prijs_verf_exclBTW = 19.90
prijs = berekenPrijsInclBTW(prijs_verf_exclBTW)
print("Prijs inclusief BTW is:", prijs)
